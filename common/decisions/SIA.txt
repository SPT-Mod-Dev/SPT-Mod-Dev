SIA_limited_war_cat = {
	SIA_start_limited_war_indochina = {

	    target_root_trigger = {
	       original_tag = SIA
	    }
	    visible = {
	    	has_completed_focus = SIA_recover_indochina
	    }
	    target_array = ROOT.indochina_controllers
	    target_trigger = {
    		NOT = {
    			is_in_array = {
    				array = previous_limited_war_targets value = FROM
    			}
    		}

    		is_in_array = {
    			array = ROOT.indochina_controllers
    			value = FROM
    		}

			any_of_scopes = {
				array = ROOT.indochina_states
				is_controlled_by = FROM
			}
	    }

	    available = {
	    	NOT = {
	    		is_ally_with = FROM
    			has_war_with = FROM
    			has_non_aggression_pact_with = FROM
    			FROM = { is_guaranteed_by = ROOT }
	    	}

    		custom_trigger_tooltip = {
    			tooltip = is_not_engaged_in_a_limited_war
	    		NOT = { has_variable = limited_war_target } 
    		}
	    }

	    highlight_states = {
	        highlight_states_trigger = {
		    	is_in_array = {
		    		array = ROOT.indochina_states
		    		value = THIS
		    	}
		    	is_controlled_by = FROM

	        }
	    }

	    cost = 20
	    complete_effect = {
	    	custom_effect_tooltip = SIA_start_limited_war_indochina_tt_main
	    	clear_variable = pre_limited_war_faction_leader

	    	if = {
	    		limit = {
	    			is_in_faction = yes
	    			is_faction_leader = no
	    		}

	    		faction_leader = { set_variable = { ROOT.pre_limited_war_faction_leader = THIS } }
	    		leave_faction = yes
	    		custom_effect_tooltip = SIA_start_limited_war_rejoin_faction
	    		custom_effect_tooltip = newline_tt

	    	}

	    	custom_effect_tooltip = newline_tt
	    	set_rule = { can_join_factions = no }

	    	custom_effect_tooltip = SIA_start_limited_war_can_join_faction
	    	#General cleanup
	    	clear_array = limited_war_states
	    	clear_variable = limited_war_target 
	    	clear_array = limited_war_target_allies

	    	#Save all states that is included in the limited war
	    	for_each_scope_loop = {
	    		array = indochina_states
	    		if = {
	    			limit = { is_controlled_by = FROM }
	
	    			add_to_array = { array = ROOT.limited_war_states value = THIS }
	    		}
