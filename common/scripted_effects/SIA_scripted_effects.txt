SIA_refresh_indochina_controllers = {
	clear_array = SIA.indochina_controllers
	
	for_each_scope_loop = {
		array = SIA.indochina_states
		if = {
			limit = {
				NOT = {
					controller = {
						is_in_array = { array = SIA.indochina_controllers value = THIS }
					}
				}
			}

			controller = {
				add_to_array = { array = SIA.indochina_controllers value = THIS }
			}
		}
	}
}

#Reset all variables related to limited wars
SIA_clear_limited_war_variables = {
	hidden_effect = {
		clear_variable = SIA.limited_war_lost_state_concession
		clear_array = SIA.limited_war_states
		clear_array = SIA.limited_war_target_allies
		clear_variable = SIA.pre_limited_war_faction_leader 
		clear_variable = SIA.limited_war_target
		SIA_refresh_indochina_controllers = yes
		SIA_refresh_malaysia_burma_controllers = yes
		set_rule = { can_join_factions = yes }
	}
}

#Re-Ads Siam to the old faction it was in pre-limited war if it was in one
SIA_re_add_to_old_faction = {
	if = {
		limit = {
			has_variable = pre_limited_war_faction_leader	
			NOT = { has_war_with = var:pre_limited_war_faction_leader }
			var:pre_limited_war_faction_leader = {
				exists = yes
				is_subject = no
				has_capitulated = no
				is_faction_leader = yes
			}
		}

	    set_rule = { can_join_factions = yes }
		var:pre_limited_war_faction_leader = {
			add_to_faction = ROOT
		}
	}
}

SIA_pick_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { SIA_poster_left_icon_frame = constant:propaganda.SIA.propaganda_empty_slot_index }
			}
			set_variable = { SIA_poster_left_icon_frame = SIA_new_propaganda_campaign }
			set_variable = { SIA_new_propaganda_campaign_duration_left = SIA_new_propaganda_campaign_duration }
			#Ageing posters
			set_variable = { SIA_propaganda_left_poster_to_age = SIA_poster_left_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
			hidden_effect = {
				country_event = { id = TAOG_siam_movie_campaign_events.101 days = constant:propaganda.SIA.default_propaganda_campaign_old_days }
				if = { #Debug -> Faster time
					limit = {
						has_country_flag = SIA_debug_fast_propaganda_campaigns_flag
					}
					set_country_flag = { flag = SIA_propaganda_left_days_remaining_flag days = 5 }

check_cheating_before_civil_war = {
	if = { 
		limit = { ##starting divs is 4
			check_variable = { num_armies < 4 }
			is_ai = no
		} 

		if = { #little cheating
			limit = {
				amount_manpower_in_deployment_queue > 0
				amount_manpower_in_deployment_queue < 1999
			}

			set_country_flag = little_cheater
		}
		else_if = { #more cheating
			limit = {
				amount_manpower_in_deployment_queue > 1999
			}
			set_country_flag = cheater
		}
	}
}

			}
		}
		start_civil_war = {
			ideology = fascism
			ruling_party = neutrality
			size = var:t_cw_size 
			keep_all_characters = yes
			set_variable = { ROOT.SIA_other_cw_country = THIS }
		}

	}
	else_if = {
		limit = { has_country_leader = { character = SIA_pridi_banomyong } }

			ideology = fascism
			ruling_party = democratic
			size = var:t_cw_size 
			keep_all_characters = yes
			set_variable = { ROOT.SIA_other_cw_country = THIS }
		}
	}
	else = { #fallback
		start_civil_war= {
			ideology = communism
			ruling_party = SIA
			size = var:t_cw_size 
			keep_all_characters = yes
			set_variable = { ROOT.SIA_other_cw_country = THIS }
		}
	}

	if = {
		limit = { has_variable = SIA_other_cw_country }

			
		random_army_leader = {
			limit = {
				NOT = {
					OR = {
						is_character = SIA_plaek_phibunsongkhram
						is_character = SIA_pridi_banomyong
						is_character = SIA_phraya_phahon
						is_character = SIA_phraya_songsuradet
						is_character = SIA_ananda_mahidol
						is_character = SIA_rambai_barni
						is_character = SIA_prajadhipok
						is_character = SIA_bowardej
						is_character = SIA_rangsit_prayurasakdi
						is_character = SIA_adul_aduldejajaras
					}
				}
			}
			set_nationality = ROOT.SIA_other_cw_country
			if = {
				limit = {
					PREV.var:SIA_other_cw_country = { has_government = fascism }
				}

				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = fascism_ideology
					}
				}
			}
			else_if = {
				limit = { 
					PREV.var:SIA_other_cw_country = { has_government = neutrality } 

				}
				
				add_country_leader_role = {
					promote_leader = yes
					country_leader = {
						ideology = despotism
					}
				}
			}
		}
	}
