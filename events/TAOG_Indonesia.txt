	id = INS_limited_war_events.7
	title = INS_limited_war_events.7.t
	desc = INS_limited_war_events.7.desc
	picture = GFX_report_event_SWE_psilander_affair 

	is_triggered_only = yes

	#Cede State, end war
	option = {
		name = INS_limited_war_events.7.a
		ai_chance = {
			base = 1
		}
		
		if = {
			limit = {
				FROM = { 
					OR = {
						check_variable = { lost_states_due_to_limited_war^num > 0 }	
						any_allied_country = {
							check_variable = { lost_states_due_to_limited_war^num > 0 }	
						}
					}
				}
			}

			FROM = {
				if = {
					limit = {
						check_variable = { lost_states_due_to_limited_war^num > 0 }	
					}
					for_each_scope_loop = {
						array = lost_states_due_to_limited_war
						
						transfer_state_to = PREV
					}

					clear_array = lost_states_due_to_limited_war
				}
				every_allied_country = {
					if = {
						limit = {
							check_variable = { lost_states_due_to_limited_war^num > 0 }	
						}
						for_each_scope_loop = {
							array = lost_states_due_to_limited_war


							transfer_state_to = PREV
						}

						clear_array = lost_states_due_to_limited_war
					}
				}
			}
		}
		else = {
			add_stability = -0.2
			add_war_support = -0.2
			random_core_state = {
				limit = {
					arms_factory > 0
				}
				
				add_building_construction = {
					type = arms_factory
					level = -1
					instant_build = yes
				}
			}
		}
		
		INS_peace_out = yes
		INS_re_add_to_old_faction = yes
		INS_clear_limited_war_variables = yes 



	}

	#No we would rather die I guess
	option = {
		name = INS_limited_war_events.7.b
		ai_chance = {
			base = 0
		}
		FROM = {
			country_event = { id = INS_limited_war_events.8 hours = 2 }
		}


		add_timed_idea = {
			idea = idea_SIA_military_humiliation
			days = 780
		}

		INS_re_add_to_old_faction = yes
		INS_clear_limited_war_variables = yes
	}

}
